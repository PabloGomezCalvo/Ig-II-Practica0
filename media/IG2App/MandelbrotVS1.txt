#version 330 core //compability
in vec4 vertex; //vertices a procesar
in vec2 uv0;
uniform mat4 modelViewProjMat;
out vec2 vUv0;
out vec2 vUv1;

uniform float time;



void main(){
	vUv0 = uv0;
	vUv1 = uv0;
	vUv0.s -= (0.5 - time);
	vUv0.t = ( uv0.t - 0.5);
	vUv0.s *= (0.5 * time);
	vUv0.t *= ( 0.5 * time);
	gl_Position = modelViewProjMat * vertex;
}
