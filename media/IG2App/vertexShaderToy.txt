#version 330 core // learnopengl.com
/* layout (location = 0) */ in vec4 aPos; // organización de
/* layout (location = 1) */ in vec3 aNormal; // los datos en la
/* layout (location = 2) */ in vec2 aTexCoords; // malla. Layout no
out vec3 FragPos; // en world space // lo usamos en Ogre
out vec3 Normal; // en world space
out vec2 TexCoords;
uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;
void main() {
FragPos = vec3(model * aPos); // en world space
Normal = mat3(transpose(inverse(model))) * aNormal;
TexCoords = aTexCoords;
gl_Position = projection * view * vec4(FragPos, 1.0);
}
